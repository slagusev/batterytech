488  void rotate(T ax, T ay, T az)
489  {
490  T a = cos(DEG2RAD(ax));
491  T b = sin(DEG2RAD(ax));
492  T c = cos(DEG2RAD(ay));
493  T d = sin(DEG2RAD(ay));
494  T e = cos(DEG2RAD(az));
495  T f = sin(DEG2RAD(az));
496  T nx = c * e * x - c * f * y + d * z;
497  T ny = (a * f + b * d * e) * x + (a * e - b * d * f) * y - b * c * z;
498  T nz = (b * f - a * d * e) * x + (a * d * f + b * e) * y + a * c * z;
499  x = nx; y = ny; z = nz;
500  }
