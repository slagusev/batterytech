835  T deltaX = right - left;
836  T deltaY = top - bottom;
837  T deltaZ = farZ - nearZ;
838  // constructed with identity
839  Matrix4<T> frust;
840 
841  if ( (nearZ <= 0.0f) || (farZ <= 0.0f) ||
842  (deltaX <= 0.0f) || (deltaY <= 0.0f) || (deltaZ <= 0.0f) )
843  return;
844 
845  frust.data[0] = 2.0f * nearZ / deltaX;
846  frust.data[1] = frust.data[2] = frust.data[3] = 0.0f;

849  frust.data[4] = frust.data[6] = frust.data[7] = 0.0f;
850 
851  frust.data[8] = (right + left) / deltaX;
852  frust.data[9] = (top + bottom) / deltaY;
853  frust.data[10] = -(nearZ + farZ) / deltaZ;
854  frust.data[11] = -1.0f;
