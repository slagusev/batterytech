109 template<typename K, typename V>
110 HashTable<K, V>* HashTable<K, V>::operator=(HashTable<K, V> *table) {
111  Bucket *bucket = NULL;
112  this->clear();
113  // Iterate over all the buckets in <table>, copying over each entry.
114  for (int i = 0; i < table->hashArraySize; i++) {
115  bucket = table->buckets[i];
116  while (bucket != NULL) {
117  put(bucket->key, bucket->data);
118  bucket = bucket->next;
119  }
120  }
121  return this;
122 }

184 template<typename K, typename V>
185 typename HashTable<K, V>::Bucket* HashTable<K, V>::findBucket(const K &key, bool remove) const {
186  int hashValue = hash(key);
187  Bucket *bucket = buckets[hashValue];
188  Bucket *lastBucket = NULL;
189  while (bucket != NULL) {
190  if (bucket->key == key) {
191  if (remove) {
192  //relink
193  if (bucket->next) {
194  if (lastBucket) {
195  lastBucket->next = bucket->next;
196  } else {
197  buckets[hashValue] = bucket->next;
198  }
199  } else if (lastBucket) {
200  lastBucket->next = NULL;
201  } else {
202  // no next, no last, is only bucket in hash array element
203  buckets[hashValue] = NULL;
204  }
205  }
206  return bucket;
207  }
208  lastBucket = bucket;
209  bucket = bucket->next;
210  }
211  return NULL;
212 }

214 template<typename K, typename V>
215 V HashTable<K, V>::getNext(Iterator &iterator) const {
216  // Visit next in current bucket, if any
217  if (iterator.bucket && iterator.bucket->next) {
218  iterator.bucket = iterator.bucket->next;
219  iterator.entriesVisited++;
220  iterator.hasNext = iterator.entriesVisited < entries;
221  iterator.key = iterator.bucket->key;
222 
223  return iterator.bucket->data;
224  }
225 
226  // Visit the remaining buckets
227  for (int i = iterator.index + 1; i < hashArraySize; ++i) {
228  Bucket *bucket = buckets[i];
229  iterator.index = i;
230  iterator.bucket = bucket;
231  if (!bucket)
232  continue;
233 
234  iterator.entriesVisited++;
235  iterator.hasNext = iterator.entriesVisited < entries;
236  iterator.key = iterator.bucket->key;
237  return iterator.bucket->data;
238  }
239 
240  iterator.hasNext = false;
241  return nullReturnVal;
242 }

362 template<typename V>
363 int StrHashTable<V>::hash(const char* key) const {
364  unsigned int hash, i, len;
365  len = strlen(key);
366  for(hash = i = 0; i < len; ++i) {
367  hash += key[i];
368  hash += (hash << 10);
369  hash ^= (hash >> 6);
370  }
371  hash += (hash << 3);
372  hash ^= (hash >> 11);
373  hash += (hash << 15);
374  return (int) (hash % HashTable<char*, V>::hashArraySize);
375 }

377 template<typename V>
378 StrHashTable<V>* StrHashTable<V>::put(const char* key, const V &data) {
379  if (((double) HashTable<char*, V>::entries / HashTable<char*, V>::hashArraySize) >= HashTable<char*, V>::THRESHOLD) {
380  HashTable<char*, V>::grow();
381  }
382  int hashValue = hash(key);
383  HashTable<char*, V>::entries++;
384  char *safeKey = new char[strlen(key)+1];
385  strcpy(safeKey, key);
386  typename HashTable<char*, V>::Bucket *bucket = new typename HashTable<char*, V>::Bucket((char*)safeKey, data);
387  /*
388  If there are already buckets in the bucket list for the given hash
389  value, prepend the new one to the existing list. Regardless,
390  update the Bucket pointer in the hash array.
391  */
392  if (HashTable<char*, V>::buckets[hashValue] != NULL) {
393  bucket->next = HashTable<char*, V>::buckets[hashValue];
394  }
395  HashTable<char*, V>::buckets[hashValue] = bucket;
396  return this;
397 }
